```python
import aspose.slides as slides

    with slides.Presentation("YOUR_DOCUMENT_DIRECTORY/welcome-to-powerpoint.pptx") as presentation:
        # Ο κώδικας για τις μεταβάσεις θα προστεθεί εδώ
```


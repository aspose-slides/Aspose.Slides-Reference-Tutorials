### คู่มือการใช้งาน
#### การสร้างและการกำหนดค่าจุดหัวข้อย่อยในย่อหน้า
มาสำรวจวิธีการสร้างจุดหัวข้อแบบกำหนดเองโดยใช้ Aspose.Slides สำหรับ .NET กัน

**ขั้นตอนที่ 1: การเริ่มต้นการนำเสนอของคุณ**
สร้างอินสแตนซ์ใหม่ของการนำเสนอของคุณซึ่งจะทำหน้าที่เป็นฐานสำหรับการเพิ่มสไลด์และเนื้อหา

```csharp
using (Presentation pres = new Presentation())
{
    // การเข้าถึงสไลด์แรก
    ISlide slide = pres.Slides[0];

    // การเพิ่ม AutoShape ของชนิดสี่เหลี่ยมผืนผ้าเพื่อเก็บข้อความ
    IAutoShape aShp = slide.Shapes.AddAutoShape(ShapeType.Rectangle, 200, 200, 400, 200);
```

**ขั้นตอนที่ 2: การเข้าถึงและการกำหนดค่ากรอบข้อความ**
ขั้นตอนถัดไปคือการกำหนดค่ากรอบข้อความภายในรูปร่างของคุณโดยการลบเนื้อหาเริ่มต้น

```csharp
    // การเข้าถึงกรอบข้อความของรูปร่างอัตโนมัติที่สร้างขึ้น
    ITextFrame txtFrm = aShp.TextFrame;

    // การลบข้อความย่อหน้าที่มีอยู่ตามค่าเริ่มต้น
    txtFrm.Paragraphs.RemoveAt(0);
```

**ขั้นตอนที่ 3: การสร้างจุดสัญลักษณ์**
สร้างจุดหัวข้อแรกของคุณโดยใช้สัญลักษณ์โดยตั้งค่าตัวเลือกการจัดรูปแบบต่างๆ

```csharp
    // การสร้างและกำหนดค่าย่อหน้าหัวข้อย่อยแรกด้วยสัญลักษณ์
    Paragraph para = new Paragraph();

    // การตั้งค่าชนิดกระสุนเป็นสัญลักษณ์
    para.ParagraphFormat.Bullet.Type = BulletType.Symbol;

    // การใช้อักขระ Unicode สำหรับสัญลักษณ์หัวข้อย่อย
    para.ParagraphFormat.Bullet.Char = Convert.ToChar(8226);

    // การเพิ่มข้อความและปรับแต่งรูปลักษณ์
    para.Text = "Welcome to Aspose.Slides";
    para.ParagraphFormat.Indent = 25; // การเยื้องจุดหัวข้อย่อย

    // การปรับแต่งสีกระสุน
    para.ParagraphFormat.Bullet.Color.ColorType = ColorType.RGB;
    para.ParagraphFormat.Bullet.Color.Color = Color.Black;
    para.ParagraphFormat.Bullet.IsBulletHardColor = NullableBool.True;

    // การกำหนดความสูงของกระสุน
    para.ParagraphFormat.Bullet.Height = 100;

    // การเพิ่มย่อหน้าลงในกรอบข้อความ
    txtFrm.Paragraphs.Add(para);
```

**ขั้นตอนที่ 4: การสร้างจุดหัวข้อแบบมีหมายเลข**
กำหนดค่าประเภทที่สองของจุดหัวข้อโดยใช้รูปแบบตัวเลข

```csharp
    // การสร้างและกำหนดค่าจุดหัวข้อย่อยที่สองด้วยรูปแบบที่มีหมายเลข
    Paragraph para2 = new Paragraph();

    // การตั้งค่าชนิดกระสุนเป็น NumberedBullet
    para2.ParagraphFormat.Bullet.Type = BulletType.Numbered;

    // ใช้สัญลักษณ์แสดงหัวข้อย่อยที่มีหมายเลขเฉพาะ
    para2.ParagraphFormat.Bullet.NumberedBulletStyle = 
        NumberedBulletStyle.BulletCircleNumWDBlackPlain;

    // การเพิ่มข้อความและปรับแต่งรูปลักษณ์
    para2.Text = "This is a numbered bullet";
    para2.ParagraphFormat.Indent = 25; // การตั้งค่าการเยื้องสำหรับจุดหัวข้อย่อยที่สอง

    // ปรับแต่งสีของกระสุนให้คล้ายกับกระสุนแรก
    para2.ParagraphFormat.Bullet.Color.ColorType = ColorType.RGB;
    para2.ParagraphFormat.Bullet.Color.Color = Color.Black;
    para2.ParagraphFormat.Bullet.IsBulletHardColor = NullableBool.True;

    // การกำหนดความสูงของหัวกระสุนสำหรับหัวกระสุนที่มีหมายเลข
    para2.ParagraphFormat.Bullet.Height = 100;

    // การเพิ่มย่อหน้าที่สองลงในกรอบข้อความ
    txtFrm.Paragraphs.Add(para2);
```

**ขั้นตอนที่ 5: บันทึกการนำเสนอของคุณ**
สุดท้ายให้บันทึกการนำเสนอของคุณไปยังไดเร็กทอรีที่ระบุ

```csharp
    // การกำหนดเส้นทางไดเรกทอรีเอาท์พุต
    string outputDir = "YOUR_OUTPUT_DIRECTORY";

    // บันทึกการนำเสนอเป็นไฟล์ PPTX
    pres.Save(outputDir + "/Bullet_out.pptx", SaveFormat.Pptx);
}
```

#### การจัดการเส้นทางไฟล์และไดเรกทอรี
ตรวจสอบให้แน่ใจว่าแอปพลิเคชันของคุณจัดการเส้นทางไฟล์อย่างถูกต้องโดยตรวจสอบดูว่ามีไดเร็กทอรีอยู่หรือไม่ก่อนที่จะบันทึกไฟล์

```csharp
using System.IO;

// กำหนดเอกสารและไดเรกทอรีผลลัพธ์ของคุณ
string dataDir = "YOUR_DOCUMENT_DIRECTORY";
string outputDir = "YOUR_OUTPUT_DIRECTORY";

// ตรวจสอบว่าไดเรกทอรีเอาท์พุตมีอยู่หรือไม่ หากไม่มีให้สร้างขึ้นใหม่
bool isExists = Directory.Exists(outputDir);
if (!isExists)
{
    // สร้างไดเรกทอรี
    Directory.CreateDirectory(outputDir);
}
```
